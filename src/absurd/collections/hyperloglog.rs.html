1use core::hash::{BuildHasher, Hash, Hasher};
2
3// we use 30 bits of the hash output
4// 14 bits are used for indexing 16384 registers, providing 1.04/sqrt(16384) = 0.8125% standard errors
5// 15 bits are used for estimate the numbers in a 4 bit register
6// todo: 15 bits are a bit insufficient, but more bits makes packing more challenging. Need a bitvec helper.
7// todo: making them configurable after generic const expr stablized?
8
9pub struct HyperLogLog<H: BuildHasher> {
10    build_hasher: H,
11    registers: [u8; 8192] // 2 registers are packed into a u8
12}
13
14#[cfg(feature = "std")]
15impl Default for HyperLogLog<std::hash::RandomState> {
16    fn default() -> Self {
17        Self::new(Default::default())
18    }
19}
20
21impl<H: BuildHasher> HyperLogLog<H> {
22    pub fn new(build_hasher: H) -> Self {
23        Self {
24            build_hasher,
25            registers: [0; 8192]
26        }
27    }
28
29    pub fn insert(&mut self, v: impl Hash) {
30        let x = {
31            let mut hasher = self.build_hasher.build_hasher();
32            v.hash(&mut hasher);
33            hasher.finish()
34        };
35
36        let j = (x >> 15) & 0b0011_1111_1111_1111;
37        let w = x as u16 & 0b0111_1111_1111_1111; // 15 bits
38        let rho = w.leading_zeros() as u8; // at least one for more accurate linear counting
39
40        let register = &mut self.registers[(j >> 1) as usize];
41        let shift = if j & 1 == 1 { 0 } else { 4 };
42        let mask = 0b1111 << shift;
43        let old = (*register & mask) >> shift;
44        let new = old.max(rho);
45        *register = (*register & !mask) | (new << shift);
46    }
47
48    pub fn count(&self) -> usize {
49        let alpha = 0.7213 / (1.0 + 1.079 / 16384.);
50
51        let z = (0..16384).map(|j| {
52            let register = &self.registers[(j >> 1) as usize];
53            let shift = if j & 1 == 1 { 0 } else { 4 };
54            let mask = 0b1111 << shift;
55            let r = (*register & mask) >> shift;
56            1. / 2i32.pow(r as _) as f32
57        }).sum::<f32>();
58
59        let mut raw = alpha * 16384. * 16384. / z;
60
61        #[cfg(feature = "std")]
62        if raw <= 2.5 * 16384. { // linear counting for small cardinality.
63            let zeros = self.registers.iter().map(|&r| {
64                (r & 0b0000_1111 == 0) as i32 + (r & 0b1111_0000 == 0) as i32
65            }).sum::<i32>();
66            raw = 16384. * (16384. / zeros as f32).ln()
67        }
68
69        raw as _
70    }
71}

2pub trait ReadExt: std::io::Read {
3    /// Allocate a `Vec<u8>` and read all bytes until EOF in this source into it.
4    fn read_to_end_alloc(&mut self) -> std::io::Result<Vec<u8>> {
5        self.read_to_end_alloc_with_capacity(0)
6    }
7
8    /// Allocate a `Vec<u8>` with specified initial capacity and read all bytes until EOF in this source into it.
9    fn read_to_end_alloc_with_capacity(&mut self, capacity: usize) -> std::io::Result<Vec<u8>> {
10        let mut buf = Vec::with_capacity(capacity);
11        self.read_to_end(&mut buf)?;
12        Ok(buf)
13    }
14
15    /// Allocate a `String` and read all bytes until EOF in this source into it.
16    fn read_to_string_alloc(&mut self) -> std::io::Result<String> {
17        self.read_to_string_alloc_with_capacity(0)
18    }
19
20    /// Allocate a `Vec<u8>` with specified initial capacity and read all bytes until EOF in this source into it.
21    fn read_to_string_alloc_with_capacity(&mut self, capacity: usize) -> std::io::Result<String> {
22        let mut buf = String::with_capacity(capacity);
23        self.read_to_string(&mut buf)?;
24        Ok(buf)
25    }
26
27    /// Allocate a `Box<[u8]>` and read exactly `n` bytes in this source into it.
28    fn read_exact_alloc(&mut self, n: usize) -> std::io::Result<Box<[u8]>> {
29        // (read_buf #78485)
30        // let buf: Box<[_]> = Box::new_uninit_slice(n);
31        let mut buf = vec![0; n].into_boxed_slice();
32        self.read_exact(&mut buf)?;
33        Ok(buf)
34    }
35}

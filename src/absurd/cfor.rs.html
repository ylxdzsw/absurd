1#[macro_export]
2macro_rules! cfor {
3    (; $($rest: tt)*) => { cfor!((); $($rest)*) };
4    ($($init: stmt),+; ; $($rest: tt)*) => { cfor!($($init),+; true; $($rest)*) };
5    ($($init: stmt),+; $cond: expr; ; $body: block) => { cfor!{$($init),+; $cond; (); $body} };
6
7    ($($init: stmt),+; $cond: expr; $($step: expr),+; $body: block) => {{
8        let mut _steped = true;
9        $($init)+
10        loop {
11            if _steped {
12                if !$cond {
13                    break
14                }
15                _steped = false;
16                $body
17            }
18
19            _steped = true;
20            $($step;)+
21        }
22    }};
23}
